// --- [속도 개선 및 가로형 구조 완벽 대응] 세종시 강수량 직접 스크래핑 --- async function fetchDetailRain() { const gridEl = document.getElementById('rain-grid'); const avgRainEl = document.getElementById('avg-rain-val'); // 로딩 중 시각적 피드백 제공 (체감 속도 향상) gridEl.innerHTML = `
세종시 강우량망 연결 중...
`; avgRainEl.innerText = `수신중`; // 1분 단위로만 갱신하여 서버 과부하 방지 및 속도 최적화 const timeMin = Math.floor(Date.now() / 60000); const targetUrl = 'https://www.sejong.go.kr/prog/safetyData/safety/sub02_02/rain/rainfallDailyList.do?_=' + timeMin; // 가장 속도가 빠르고 텍스트를 직접 반환하는 프록시 서버로 교체 const proxyUrl = `https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(targetUrl)}`; const locs = [ { name: '조치원', lat: 36.601, lon: 127.301 }, { name: '연서', lat: 36.560, lon: 127.270 }, { name: '전의', lat: 36.670, lon: 127.200 }, { name: '전동', lat: 36.630, lon: 127.260 }, { name: '소정', lat: 36.720, lon: 127.150 }, { name: '부강', lat: 36.520, lon: 127.370 }, { name: '연동', lat: 36.560, lon: 127.320 } ]; const lats = locs.map(l => l.lat).join(','); const lons = locs.map(l => l.lon).join(','); const tempUrl = `https://api.open-meteo.com/v1/forecast?latitude=${lats}&longitude=${lons}¤t=temperature_2m&timezone=Asia%2FTokyo`; try { const [rainRes, tempRes] = await Promise.all([ fetch(proxyUrl), fetch(tempUrl) ]); // JSON 포장을 벗기지 않고 Text 자체를 바로 파싱하여 속도 대폭 향상 const htmlText = await rainRes.text(); const parser = new DOMParser(); const doc = parser.parseFromString(htmlText, 'text/html'); const sejongRain = {}; const rows = doc.querySelectorAll('table tr'); let headerRow = []; let valueRow = []; rows.forEach(row => { const cells = Array.from(row.querySelectorAll('th, td')).map(c => c.textContent.trim()); if (cells.some(c => c.includes('조치원'))) headerRow = cells; if (cells.some(c => c.includes('합계'))) valueRow = cells; }); if (headerRow.length > 0 && valueRow.length > 0) { locs.forEach(loc => { const idx = headerRow.findIndex(h => h.includes(loc.name)); if (idx !== -1 && idx < valueRow.length) { const valStr = valueRow[idx].replace(/,/g, ''); const num = parseFloat(valStr); sejongRain[loc.name] = isNaN(num) ? 0 : num; } }); } const tempData = await tempRes.json(); const tempResults = Array.isArray(tempData) ? tempData : [tempData]; let html = ''; let totalRain = 0; let count = 0; locs.forEach((loc, idx) => { const todayRain = sejongRain[loc.name] !== undefined ? sejongRain[loc.name] : 0; const currentTemp = tempResults[idx]?.current?.temperature_2m ?? '-'; totalRain += todayRain; count++; const valClass = todayRain > 0 ? 'rain-val wet' : 'rain-val'; const rainColorStyle = todayRain > 0 ? '' : 'color: #888;'; html += `
${loc.name}
${todayRain}mm
${currentTemp}℃
`; }); html += `
출처
세종시망
실시간
`; gridEl.innerHTML = html; const avgRain = count > 0 ? (totalRain / count).toFixed(1) : '-'; avgRainEl.innerText = `${avgRain} mm`; } catch (e) { console.error("데이터 병합 로드 에러:", e); avgRainEl.innerText = "로딩 지연"; gridEl.innerHTML = `
데이터 수신 지연. 우측 상단 새로고침을 눌러주세요.
`; } }
